// if SD card is not proper, give error sd card not mounted
// if mount is successful, if file is not there, display file not found error
// spi.h,sd.h,fs.h are built in libraries

#include <SPI.h>
#include <TFT_eSPI.h>

File myFile;
TFT_eSPI tft = TFT_eSPI();

void setup() {
  tft.init();
  tft.setRotation(3); // Adjust the rotation if necessary (0 to 3)
  
  
  Serial.begin(115200);
  tft.fillScreen(TFT_WHITE); // Fill the screen with black color
  tft.setTextColor(TFT_BLACK); // Set text color to white
  tft.setTextSize(2); // Set text size to 2
  
  tft.setCursor(2, 2); // Set the cursor position
  
  delay(1000);
  while (!SD.begin()) {
    tft.println("Card Mount Failed");
    delay(1000);
  }
  tft.println("setup success");
   
   tft.println("Yaare Koogadali Hoore Horadali");
   myFile = SD.open("/rvce.txt");
   if (!myFile) {
      tft.println("Couldn't open file");
      }
   else{
      tft.print("Welcome! ");
      while (myFile.available()) { //Print out the content of that file
      tft.write(myFile.read());
      }
      }
}
