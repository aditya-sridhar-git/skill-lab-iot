char readKB(void)
{
  char keys[] = "147*2580369#ABCDNF";  // N = Nokey, F = Fail, table keys indicated column wise
  
  uint8_t idx;
  do{
  idx = keyPad.getKey();
  if (  keys[idx] != 'N') 
     {
      delay(300);
      return(keys[idx]);
     }
  }while(1);
 
}

  }
  // result = 0;

}



void stepper(){
   int steps=500;
    while(1)
    { 
    Wire.beginTransmission(PCF8574_U8);
    Wire.write(0b11110111);
    Wire.endTransmission();
    delay(10);
    steps--;
    if(steps==0)
    break;
    Wire.beginTransmission(PCF8574_U8);
    Wire.write(0b11111011);
    Wire.endTransmission();
    delay(10);
    steps--;
    if(steps==0)
    break;
    Wire.beginTransmission(PCF8574_U8);
    Wire.write(0b11111101);
    Wire.endTransmission();
    delay(10);
    steps--;
    if(steps==0)
    break;
    // digitalWrite(2, HIGH);

void runservo()
{
  int pwm0;
  // Initialize PCA9685
  pca9685.begin();
  // Set PWM Frequency to 50Hz
  pca9685.setPWMFreq(50);

  for (int posDegrees = 0; posDegrees <= 180; posDegrees++) {
 
    // Determine PWM pulse width
    pwm0 = map(posDegrees, 0, 180, SERVOMIN, SERVOMAX);
    // Write to PCA9685
    pca9685.setPWM(SER0, 0, pwm0);
    // Print to serial monitor
    Serial.print("Motor 0 = ");
    Serial.println(posDegrees);
    delay(30);
  }

   for (int posDegrees = 180; posDegrees >= 0; posDegrees--) {
 
    // Determine PWM pulse width
    pwm0 = map(posDegrees, 0, 180, SERVOMIN, SERVOMAX);
    // Write to PCA9685
    pca9685.setPWM(SER0, 0, pwm0);
    // Print to serial monitor
    Serial.print("Motor 0 = ");
    Serial.println(posDegrees);
    delay(30);
  }

}

void stepper2()
{
  digitalWrite(DIR, HIGH);
  Serial.println("Spinning Clockwise...");
  
  for(int i = 0; i<steps_per_rev; i++)
  {
    digitalWrite(STEP, HIGH);
    delayMicroseconds(20000);
    digitalWrite(STEP, LOW);
    delayMicroseconds(20000);
  }
  delay(500); 
  
  digitalWrite(DIR, LOW);
  Serial.println("Spinning Anti-Clockwise...");

  for(int i = 0; i<steps_per_rev; i++)
  {
    digitalWrite(STEP, HIGH);
    delayMicroseconds(20000);
    digitalWrite(STEP, LOW);
    delayMicroseconds(20000);
  }
  delay(500);
}
