//Upload code. If the buzzer is producing sounds, the connections are fine. 
//Ensure the appropriate combiner pin of GPIO25 is connected.
const int buzzer = 25; 
int playing = 0;
void tone(byte pin, int freq) {
  ledcSetup(0, 2000, 8); // setup beeper
  ledcAttachPin(pin, 0); // attach beeper
  ledcWriteTone(0, freq); // play tone
  playing = pin; // store pin
}
void noTone() {
  tone(playing, 0);
}

void setup(){
  pinMode(buzzer, OUTPUT); 
}
void loop(){
  tone(buzzer, 5000); // Send 1KHz sound signal...
  delay(1000);        // ...for 1 sec
  noTone();     // Stop sound...
  delay(1000);        // ...for 1sec
}
